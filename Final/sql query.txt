-- 1. Create the Database
CREATE DATABASE IF NOT EXISTS mydatabest;
USE mydatabest;

-- 2. Create Tables

-- Table: Students
CREATE TABLE IF NOT EXISTS students (
    id VARCHAR(20) PRIMARY KEY,
    fname VARCHAR(100) NOT NULL,
    lname VARCHAR(100) NOT NULL,
    mname VARCHAR(100) DEFAULT NULL,
    email VARCHAR(100) UNIQUE,
    course VARCHAR(50) NOT NULL,
    course_level INT NOT NULL,
    password VARCHAR(255) NOT NULL,
    address TEXT,
    role ENUM('admin', 'student') DEFAULT 'student',
    remaining_session INT DEFAULT 30,
    reward_points INT DEFAULT 0,
    tasks_completed INT DEFAULT 0,
    total_rewards_earned INT DEFAULT 0,
    profile_picture VARCHAR(255) DEFAULT 'default_profile.png',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Table: Sit-in Records
CREATE TABLE IF NOT EXISTS sitin_records (
    id INT AUTO_INCREMENT PRIMARY KEY,
    student_id VARCHAR(20) NOT NULL,
    purpose VARCHAR(100) NOT NULL,
    lab VARCHAR(50) NOT NULL,
    pc_number VARCHAR(50) NOT NULL,
    date_created DATETIME DEFAULT CURRENT_TIMESTAMP,
    date_ended DATETIME DEFAULT NULL,
    end_time TIME DEFAULT NULL,
    extension_count INT DEFAULT 0,
    status ENUM('pending', 'active', 'completed') DEFAULT 'pending',
    FOREIGN KEY (student_id) REFERENCES students(id) ON DELETE CASCADE
);

-- Table: Reservations
-- Note: Includes start_time and end_time for accurate overlap checking
CREATE TABLE IF NOT EXISTS reservations (
    id INT AUTO_INCREMENT PRIMARY KEY,
    student_id VARCHAR(20) NOT NULL,
    lab VARCHAR(50) NOT NULL,
    pc_number VARCHAR(50) NOT NULL,
    purpose VARCHAR(100) NOT NULL,
    reservation_datetime DATETIME NOT NULL,
    start_time TIME NOT NULL,
    end_time TIME NOT NULL,
    remaining_session INT DEFAULT 30,
    status ENUM('pending', 'confirmed', 'declined') DEFAULT 'pending',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (student_id) REFERENCES students(id) ON DELETE CASCADE
);

-- Table: Notifications
CREATE TABLE IF NOT EXISTS notifications (
    id INT AUTO_INCREMENT PRIMARY KEY,
    student_id VARCHAR(20) NOT NULL,
    title VARCHAR(100) NOT NULL,
    message TEXT,
    type VARCHAR(50) DEFAULT 'info',
    is_read TINYINT(1) DEFAULT 0,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (student_id) REFERENCES students(id) ON DELETE CASCADE
);

-- Table: Feedbacks
CREATE TABLE IF NOT EXISTS feedbacks (
    id INT AUTO_INCREMENT PRIMARY KEY,
    student_id VARCHAR(20) NOT NULL,
    record_id INT NOT NULL,
    message TEXT,
    status ENUM('pending', 'resolved') DEFAULT 'pending',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (student_id) REFERENCES students(id) ON DELETE CASCADE,
    FOREIGN KEY (record_id) REFERENCES sitin_records(id) ON DELETE CASCADE
);

-- Table: Announcements
CREATE TABLE IF NOT EXISTS announcements (
    id INT AUTO_INCREMENT PRIMARY KEY,
    admin_id VARCHAR(20) NOT NULL,
    content TEXT NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (admin_id) REFERENCES students(id) ON DELETE CASCADE
);
